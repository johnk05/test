python     def measure_student_improvement(student_id):
         baseline_scores = get_scores_before_recommendations(student_id)
         intervention_scores = get_scores_after_recommendations(student_id)
         t_stat, p_value = paired_ttest(baseline_scores, intervention_scores)
         effect_size = cohens_d(baseline_scores, intervention_scores)
         return {
             'mean_improvement': mean(intervention_scores) - mean(baseline_scores),
             'p_value': p_value,
             'significant': p_value < 0.05,
             'effect_size': effect_size
         }
